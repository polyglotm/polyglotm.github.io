(define (f a b c)
  (+
   (square (maxnum a b c)) 
   (square (middlenum a b c)) 
  )
 )
(define (maxnum a b c)
  (cond 
    ((and (> a b) (> a c)) a)
    ((and (> b a) (> b c)) b)
    ((and (> c a) (> c b)) c)
   )
  )
(define (middlenum a b c)
   (cond
     ((and (> a b)(> b c)) b)
     ((and (> b c) (> c a)) c)
     ((and (> c a) (> a b)) a)
     )
   ) 
  
(define (square a )
  (* a a)
  )
(f 5 3 1)
